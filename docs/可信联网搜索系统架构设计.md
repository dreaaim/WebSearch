# 可信联网搜索系统架构设计

## 一、系统概述

### 1.1 设计目标

本系统旨在构建一个**可信联网搜索**基础设施，为AI应用提供实时、可信、可验证的网络信息获取能力。区别于传统搜索系统仅返回信息列表，本系统强调：

- **信源可信**: 通过白名单/灰名单/黑名单机制对信源进行分层
- **事实碰撞解决**: 当多个信源信息冲突时，按照优先级规则智能裁决
- **社区共识摘要**: 汇聚相同信息，生成带来源标注的共识摘要

### 1.2 核心价值

| 能力 | 说明 |
|------|------|
| 实时联网 | 获取最新信息，突破LLM知识时效限制 |
| 信源分层 | 区分权威信源与普通信源 |
| 碰撞解决 | 智能处理信息冲突，保证可信度 |
| 共识摘要 | 聚合多源观点，识别共识与分歧 |

```python
from abc import ABC, abstractmethod
from typing import Optional
from .models import SearchResponse, SearchOptions


class SearchProvider(ABC):
    """搜索引擎抽象基类"""

    @property
    @abstractmethod
    def name(self) -> str:
        """提供商名称"""
        pass

    @property
    def supported_engines(self) -> list[str]:
        """支持的引擎列表（如 SearXNG 支持 google, bing 等）"""
        return []

    @abstractmethod
    def search(
        self,
        query: str,
        options: Optional[SearchOptions] = None
    ) -> SearchResponse:
        """
        执行搜索

        Args:
            query: 搜索查询
            options: 搜索选项

        Returns:
            SearchResponse: 标准化搜索响应
        """
        pass

    @abstractmethod
    def validate_config(self) -> bool:
        """验证配置是否正确"""
        pass
```

### 5.2 SearXNG Provider 实现

```python
import requests
from urllib.parse import urlencode
from typing import Optional
from .base import SearchProvider
from ..models import SearchResponse, SearchResult, SearchOptions, SourceType, SourceLevel, Classification


class SearXNGProvider(SearchProvider):
    """SearXNG 搜索引擎 Provider"""

    def __init__(
        self,
        base_url: str = "http://localhost:8080",
        api_key: Optional[str] = None,
        default_engines: Optional[list[str]] = None
    ):
        self.base_url = base_url.rstrip("/")
        self.api_key = api_key
        self.default_engines = default_engines or ["google", "bing"]

    @property
    def name(self) -> str:
        return "searxng"

    @property
    def supported_engines(self) -> list[str]:
        return ["google", "bing", "duckduckgo", "baidu", "yandex", "brave"]

    def search(
        self,
        query: str,
        options: Optional[SearchOptions] = None
    ) -> SearchResponse:
        options = options or SearchOptions()
        engines = options.engines or self.default_engines

        params = {
            "q": query,
            "engines": ",".join(engines),
            "format": "json",
            "limit": options.max_results,
        }

        if options.time_range:
            params["time_range"] = options.time_range

        headers = {}
        if self.api_key:
            headers["X-API-KEY"] = self.api_key

        response = requests.get(
            f"{self.base_url}/search",
            params=params,
            headers=headers,
            timeout=30
        )
        response.raise_for_status()

        data = response.json()
        results = self._parse_results(data)

        return SearchResponse(
            query=query,
            results=results,
            total_count=len(results),
            search_time=data.get("time", 0.0)
        )

    def _parse_results(self, data: dict) -> List[SearchResult]:
        """解析 SearXNG 响应"""
        results = []
        for item in data.get("results", []):
            result = SearchResult(
                title=item.get("title", ""),
                url=item.get("url", ""),
                snippet=item.get("content", ""),
                source_name=item.get("source", {}).get("name", ""),
                source_domain=self._extract_domain(item.get("url", "")),
                source_type=self._infer_source_type(item),
                source_level=self._infer_source_level(item),
            )
            results.append(result)
        return results

    def _extract_domain(self, url: str) -> str:
        """提取域名"""
        from urllib.parse import urlparse
        try:
            return urlparse(url).netloc
        except:
            return ""

    def _infer_source_type(self, item: dict) -> SourceType:
        """推断信源类型"""
        url = item.get("url", "").lower()
        if "gov.cn" in url or "gov." in url:
            return SourceType.OFFICIAL
        return SourceType.MEDIA

    def _infer_source_level(self, item: dict) -> SourceLevel:
        """推断信源层级"""
        # 简化实现，实际需要更复杂的逻辑
        return SourceLevel.MUNICIPAL

    def validate_config(self) -> bool:
        """验证配置"""
        try:
            response = requests.get(
                f"{self.base_url}/health",
                timeout=5
            )
            return response.status_code == 200
        except:
            return False
```

```python
    SEARXNG = "searxng"
    TAVILY = "tavily"

class SearchProviderFactory:
    """SearchProvider 工厂类"""

    _providers: dict[str, type[SearchProvider]] = {
        ProviderType.SEARXNG.value: SearXNGProvider,
    }

    @classmethod
    def register(cls, name: str, provider_class: type[SearchProvider]):
        """注册新的 Provider"""
        cls._providers[name] = provider_class

    @classmethod
    def create(
        cls,
        name: str,
        **kwargs
    ) -> Optional[SearchProvider]:
        """创建 Provider 实例"""
        provider_class = cls._providers.get(name.lower())
        if not provider_class:
            raise ValueError(f"Unknown provider: {name}")
        return provider_class(**kwargs)
```

### 6.2 核心实现

```python
from typing import List, Optional, Tuple
import re
from ..models import SearchResult, Classification, SourceType, SourceLevel


class SourceClassifier:
    """信源分层分类器"""

    def __init__(
        self,
        whitelist: List[dict],
        blacklist: List[dict],
        kol_thresholds: dict = None
    ):
        self.whitelist = whitelist
        self.blacklist = blacklist
        self.kol_thresholds = kol_thresholds or {
            "big": 1000000,    # 粉丝>100万
            "medium": 100000, # 粉丝>10万
            "small": 10000    # 粉丝>1万
        }

    def classify(self, result: SearchResult) -> Classification:
        """
        对单个搜索结果进行分类

        Returns:
            Classification: WHITE, GRAY, 或 BLACK
        """
        # 1. 先检查黑名单
        if self._is_blacklisted(result):
            return Classification.BLACK

        # 2. 检查白名单
        if self._is_whitelisted(result):
            return Classification.WHITE

        # 3. 灰名单：其他所有
        return Classification.GRAY

    def _is_blacklisted(self, result: SearchResult) -> bool:
        """检查是否在黑名单中"""
        for rule in self.blacklist:
            if self._match_rule(result, rule):
                return True
        return False

    def _is_whitelisted(self, result: SearchResult) -> bool:
        """检查是否在白名单中"""
        for rule in self.whitelist:
            if self._match_rule(result, rule):
                return True
        return False

    def _match_rule(self, result: SearchResult, rule: dict) -> bool:
        """匹配单条规则"""
        domain = result.source_domain.lower()

        # 精确域名匹配
        if "domain" in rule:
            if domain == rule["domain"].lower():
                return True

        # 域名后缀匹配 (如 *.gov.cn)
        if "domain_suffix" in rule:
            suffix = rule["domain_suffix"].lower()
            if domain.endswith(suffix):
                return True

        # 通配符模式匹配
        if "domain_pattern" in rule:
            pattern = rule["domain_pattern"].lower().replace("*", ".*")
            if re.match(pattern, domain):
                return True

        return False

    def classify_results(
        self,
        results: List[SearchResult]
    ) -> dict[str, List[SearchResult]]:
        """
        批量分类搜索结果

        Returns:
            {
                "white": [...],
                "gray": [...],
                "black": [...]
            }
        """
        classified = {
            "white": [],
            "gray": [],
            "black": []
        }

        for result in results:
            classification = self.classify(result)
            result.classification = classification
            classified[classification.value].append(result)

        return classified

    def infer_source_type(self, result: SearchResult) -> SourceType:
        """
        推断信源类型

        推断逻辑:
        1. 政府域名 -> OFFICIAL
        2. 知名媒体 -> MEDIA
        3. KOL (有粉丝数) -> KOL
        4. 其他 -> INDIVIDUAL
        """
        domain = result.source_domain.lower()

        # 政府机构
        government_suffixes = [".gov.cn", ".gov.", ".org.cn"]
        if any(domain.endswith(s) for s in government_suffixes):
            return SourceType.OFFICIAL

        # KOL 判断
        if result.粉丝数 and result.粉丝数 > 0:
            if result.粉丝数 >= self.kol_thresholds["small"]:
                return SourceType.KOL

        # 知名媒体模式
        media_patterns = [
            "news", "press", "media", "journal",
            "xinhuanet", "people.com", "cctv", "cCTV"
        ]
        if any(p in domain for p in media_patterns):
            return SourceType.MEDIA

        return SourceType.INDIVIDUAL

    def infer_source_level(self, result: SearchResult) -> SourceLevel:
        """
        推断信源层级

        推断逻辑:
        1. 国家顶级域名/中央媒体 -> NATIONAL
        2. 省级域名 (*.ah.cn, *.zj.cn 等) -> PROVINCIAL
        3. 市级域名 (*.gz.cn 等) -> MUNICIPAL
        4. 其他默认 -> LOCAL
        """
        domain = result.source_domain.lower()

        # 国家机构
        national_patterns = [
            "gov.cn", "xinhuanet", "people.com.cn", "cctv.com",
            "gov.hk", "gov.tw", "org.cn"
        ]
        if any(p in domain for p in national_patterns):
            return SourceLevel.NATIONAL

        # 省级域名匹配
        provincial_pattern = r"\.[a-z]{2}\.gov\.cn$"
        if re.search(provincial_pattern, domain):
            return SourceLevel.PROVINCIAL

        # 市级域名匹配
        municipal_pattern = r"\.[a-z]{3,}\.gov\.cn$"
        if re.search(municipal_pattern, domain):
            return SourceLevel.MUNICIPAL

        return SourceLevel.LOCAL
```

### 7.2 优先级规则引擎

```python

class PriorityLevel(IntEnum):

    BLACK_INDIVIDUAL = 1
    BLACK_MEDIA = 2
    GRAY_INDIVIDUAL = 3
    GRAY_MEDIA = 4
    GRAY_KOL_SMALL = 5
    GRAY_KOL_MEDIUM = 6
    GRAY_KOL_BIG = 7
    WHITE_MUNICIPAL = 8
    WHITE_PROVINCIAL = 9
    WHITE_NATIONAL_OFFICIAL = 10


class PriorityEngine:
    """优先级规则引擎"""

    def __init__(self, kol_thresholds: dict = None):
        self.kol_thresholds = kol_thresholds or {
            "big": 1000000,
            "medium": 100000,
            "small": 10000
        }

    def calculate_priority(self, result: SearchResult) -> Tuple[PriorityLevel, str]:
        """
        计算单个结果的优先级

        Returns:
            (PriorityLevel, rule_description)
        """
        classification = result.classification
        source_type = result.source_type
        source_level = result.source_level
        粉丝数 = result.粉丝数 or 0

        # === BLACK LIST ===
        if classification == Classification.BLACK:
            if source_type == SourceType.INDIVIDUAL:
                return PriorityLevel.BLACK_INDIVIDUAL, "黑名单-个人"
            return PriorityLevel.BLACK_MEDIA, "黑名单-媒体"

        # === GRAY LIST ===
        if classification == Classification.GRAY:
            if source_type == SourceType.KOL:
                if 粉丝数 >= self.kol_thresholds["big"]:
                    return PriorityLevel.GRAY_KOL_BIG, f"灰名单-KOL(粉丝{粉丝数})"
                elif 粉丝数 >= self.kol_thresholds["medium"]:
                    return PriorityLevel.GRAY_KOL_MEDIUM, f"灰名单-KOL(粉丝{粉丝数})"
                return PriorityLevel.GRAY_KOL_SMALL, f"灰名单-KOL(粉丝{粉丝数})"
            if source_type == SourceType.INDIVIDUAL:
                return PriorityLevel.GRAY_INDIVIDUAL, "灰名单-个人"
            return PriorityLevel.GRAY_MEDIA, "灰名单-媒体"

        # === WHITE LIST ===
        if classification == Classification.WHITE:
            if source_level == SourceLevel.NATIONAL:
                return PriorityLevel.WHITE_NATIONAL_OFFICIAL, "白名单-国家级机构"
            if source_level == SourceLevel.PROVINCIAL:
                return PriorityLevel.WHITE_PROVINCIAL, "白名单-省级机构"
            if source_level == SourceLevel.MUNICIPAL:
                return PriorityLevel.WHITE_MUNICIPAL, "白名单-市级机构"

        # 默认最低优先级
        return PriorityLevel.BLACK_INDIVIDUAL, "默认最低优先级"

    def resolve_collision(self, claims: List[Claim]) -> FactCollision:
        """
        解决碰撞 - 找出最可信的claim

        规则:
        1. 先按分类层级 (WHITE > GRAY > BLACK)
        2. 同层级按 source_type (OFFICIAL > KOL > MEDIA > INDIVIDUAL)
        3. 同类型按 source_level (NATIONAL > PROVINCIAL > MUNICIPAL > LOCAL)
        4. KOL按粉丝数排序
        """
        if len(claims) < 2:
            return FactCollision(
                collision_id="single",
                claims=claims,
                resolved_claim=claims[0] if claims else None,
                priority_rule_used="single_claim",
                consensus_degree=1.0
            )

        # 计算每个claim的优先级
        scored_claims = []
        for claim in claims:
            priority, rule = self.calculate_priority(claim.result)
            scored_claims.append((claim, priority, rule))

        # 按优先级排序
        scored_claims.sort(key=lambda x: x[1], reverse=True)

        # 计算共识度 (最高优先级的claim数量 / 总claim数量)
        top_priority = scored_claims[0][1]
        top_count = sum(1 for _, p, _ in scored_claims if p == top_priority)
        consensus = top_count / len(claims)

        return FactCollision(
            collision_id=self._generate_collision_id(claims),
            claims=claims,
            resolved_claim=scored_claims[0][0],
            priority_rule_used=scored_claims[0][2],
            consensus_degree=consensus
        )

    def _generate_collision_id(self, claims: List[Claim]) -> str:
        """生成碰撞ID"""
        topics = [c.statement[:50] for c in claims]
        return f"collision_{hash(tuple(topics))}"
```

### 7.3 碰撞解决器

```python
from typing import List, Optional
from .priority_engine import PriorityEngine
from .claim import ClaimExtractor
from ..models import SearchResult, FactCollision, Claim, FactCluster


class FactResolver:
    """事实碰撞解决器"""

    def __init__(self, priority_engine: PriorityEngine = None):
        self.priority_engine = priority_engine or PriorityEngine()
        self.claim_extractor = ClaimExtractor()

    def detect_and_resolve(
        self,
        results: List[SearchResult]
    ) -> List[FactCollision]:
        """
        检测并解决所有碰撞

        流程:
        1. 提取每个结果的claims
        2. 计算claim之间的相似度
        3. 识别碰撞组
        4. 对每个碰撞组进行裁决
        """
        # 1. 提取claims
        claims = []
        for result in results:
            extracted_claims = self.claim_extractor.extract(result)
            claims.extend(extracted_claims)

        # 2. 分组 - 简化实现，实际需要更复杂的聚类
        claim_groups = self._group_similar_claims(claims)

        # 3. 解决每个碰撞组
        collisions = []
        for group in claim_groups:
            if len(group) >= 2:
                collision = self.priority_engine.resolve_collision(group)
                collisions.append(collision)

        return collisions

    def _group_similar_claims(self, claims: List[Claim]) -> List[List[Claim]]:
        """
        将相似的claims分组

        简化实现: 基于关键词匹配分组
        实际应使用: 语义相似度 (embedding) + 实体识别
        """
        groups = []
        used = set()

        for i, claim in enumerate(claims):
            if i in used:
                continue

            group = [claim]
            used.add(i)

            for j, other in enumerate(claims[i+1:], start=i+1):
                if j in used:
                    continue
                if self._are_claims_similar(claim, other):
                    group.append(other)
                    used.add(j)

            groups.append(group)

        return groups

    def _are_claims_similar(self, c1: Claim, c2: Claim) -> bool:
        """判断两个claim是否描述同一事实"""
        # 简化: 基于共享关键词
        words1 = set(c1.statement.lower().split())
        words2 = set(c2.statement.lower().split())
        overlap = len(words1 & words2) / len(words1 | words2)
        return overlap > 0.6
```

```python
from typing import List, Dict, Tuple
from collections import Counter
from ..models import SearchResult, Claim, FactCluster


class SummaryGenerator:
    """摘要生成器"""

    def __init__(self, min_consensus_ratio: float = 0.6):
        self.min_consensus_ratio = min_consensus_ratio

    def generate(
        self,
        results: List[SearchResult],
        collisions: List = None
    ) -> Tuple[str, List[str], List[str]]:
        """
        生成摘要

        Returns:
            (summary, consensus_facts, disputed_facts)
        """
        # 1. 聚类
        clusters = self._cluster_results(results)

        # 2. 检测共识与分歧
        consensus_facts = []
        disputed_facts = []

        for cluster in clusters:
            is_consensus, fact = self._detect_consensus(cluster)
            if is_consensus:
                consensus_facts.append(fact)
            else:
                disputed_facts.append(fact)

        # 3. 生成摘要
        summary = self._build_summary(clusters, consensus_facts, disputed_facts)

        return summary, consensus_facts, disputed_facts

    def _cluster_results(self, results: List[SearchResult]) -> List[List[SearchResult]]:
        """
        将搜索结果按主题聚类

        实现策略:
        - 基于关键词提取主题词
        - 主题词相同的结果归为同一类
        - 考虑来源权威性加权
        """
        clusters: Dict[str, List[SearchResult]] = {}

        for result in results:
            topic = self._extract_topic(result)
            if topic not in clusters:
                clusters[topic] = []
            clusters[topic].append(result)

        return list(clusters.values())

    def _extract_topic(self, result: SearchResult) -> str:
        """
        提取结果主题

        简化实现: 使用TF-IDF提取关键词
        实际应使用: NLP实体识别 + 主题模型
        """
        # 简单实现: 合并标题和摘要，取前100字符作为主题标识
        text = f"{result.title} {result.snippet}"
        words = text.split()[:10]
        return " ".join(words)

    def _detect_consensus(
        self,
        cluster: List[SearchResult]
    ) -> Tuple[bool, str]:
        """
        检测聚类内的共识

        Returns:
            (is_consensus, consensus_fact)
        """
        if len(cluster) < 2:
            return True, cluster[0].snippet if cluster else ""

        # 提取核心声明
        statements = [r.snippet[:100] for r in cluster]

        # 简化的共识检测: 统计相似声明
        # 实际应使用: 语义相似度计算
        statement_hash = Counter(statements)
        most_common_count = statement_hash.most_common(1)[0][1]
        ratio = most_common_count / len(statements)

        if ratio >= self.min_consensus_ratio:
            return True, statement_hash.most_common(1)[0][0]

        return False, f"存在分歧: {len(cluster)}个不同观点"

    def _build_summary(
        self,
        clusters: List[List[SearchResult]],
        consensus_facts: List[str],
        disputed_facts: List[str]
    ) -> str:
        """构建摘要文本"""
        lines = []

        # 共识事实
        if consensus_facts:
            lines.append("【共识事实】")
            for fact in consensus_facts[:3]:  # 最多3条
                lines.append(f"• {fact}")
            lines.append("")

        # 争议事实
        if disputed_facts:
            lines.append("【争议/分歧】")
            for fact in disputed_facts[:3]:
                lines.append(f"• {fact}")
            lines.append("")

        # 来源统计
        total = sum(len(c) for c in clusters)
        lines.append(f"共聚合 {total} 条来源，涵盖 {len(clusters)} 个不同角度的报道。")

        return "\n".join(lines)

    def generate_with_sources(
        self,
        result: SearchResult,
        related_results: List[SearchResult]
    ) -> str:
        """
        生成带来源标注的摘要

        格式: [摘要内容](来源: source_name, 可信度: classification)
        """
        classification_emoji = {
            "white": "✅",
            "gray": "⚠️",
            "black": "❌"
        }

        emoji = classification_emoji.get(result.classification.value, "⚠️")

        citation = f"{emoji} [{result.source_name}]({result.url})"

        return f"{result.snippet}\n来源: {citation}"
```

## 九、SearchOrchestrator 编排器

### 9.1 编排器实现

```python
from typing import Optional
from .providers.base import SearchProvider
from .providers.factory import SearchProviderFactory
from .classifier.source_classifier import SourceClassifier
from .resolver.fact_resolver import FactResolver
from .summary.summary_generator import SummaryGenerator
from .models import (
    SearchOptions, SearchResponse, TrustedSearchResult,
    Classification
)


class SearchOrchestrator:
    """搜索编排器 - 核心调度中心"""

    def __init__(
        self,
        provider: SearchProvider,
        source_classifier: SourceClassifier,
        fact_resolver: FactResolver = None,
        summary_generator: SummaryGenerator = None
    ):
        self.provider = provider
        self.source_classifier = source_classifier
        self.fact_resolver = fact_resolver or FactResolver()
        self.summary_generator = summary_generator or SummaryGenerator()

    def search_with_trust(
        self,
        query: str,
        options: Optional[SearchOptions] = None
    ) -> TrustedSearchResult:
        """
        执行可信搜索

        完整流程:
        1. 调用搜索引擎获取原始结果
        2. 对结果进行信源分层
        3. 检测并解决事实碰撞
        4. 生成社区摘要
        5. 返回可信搜索结果
        """
        # === Step 1: 执行搜索 ===
        response = self.provider.search(query, options)
        results = response.results

        # === Step 2: 信源分层 ===
        classified = self.source_classifier.classify_results(results)

        # 补充信源类型和层级推断
        for result in results:
            if result.source_type is None:
                result.source_type = self.source_classifier.infer_source_type(result)
            if result.source_level is None:
                result.source_level = self.source_classifier.infer_source_level(result)

        # === Step 3: 碰撞检测与解决 ===
        collisions = self.fact_resolver.detect_and_resolve(results)

        # === Step 4: 生成摘要 ===
        summary, consensus_facts, disputed_facts = self.summary_generator.generate(
            results, collisions
        )

        # === Step 5: 组装结果 ===
        return TrustedSearchResult(
            query=query,
            response=response,
            classified_results=classified,
            collisions=collisions,
            clusters=[],  # 简化省略
            summary=summary,
            consensus_facts=consensus_facts,
            disputed_facts=disputed_facts,
            metadata={
                "provider": self.provider.name,
                "white_count": len(classified["white"]),
                "gray_count": len(classified["gray"]),
                "black_count": len(classified["black"]),
                "collision_count": len(collisions)
            }
        )

    def white_list_search(
        self,
        query: str,
        options: Optional[SearchOptions] = None
    ) -> SearchResponse:
        """
        仅白名单搜索 - 只返回高可信来源
        """
        options = options or SearchOptions()
        options.classification = Classification.WHITE

        result = self.search_with_trust(query, options)
        return SearchResponse(
            query=query,
            results=result.classified_results["white"],
            total_count=len(result.classified_results["white"]),
            search_time=result.response.search_time
        )
```

```
    - consensus_degree: 共识度
    - priority_rule_used: 使用的优先级规则
```

## 十四、总结

本架构设计文档详细阐述了**可信联网搜索系统**的完整技术方案：

1. **SearchProvider 抽象层**: 支持 SearXNG/Tavily/Exa/Bing 等多种搜索引擎，可插拔设计便于后续扩展

2. **SourceClassifier 信源分层**: 通过白名单/灰名单/黑名单机制对信源进行可信度分层

3. **FactResolver 碰撞解决**: 基于优先级规则引擎，智能解决多源信息冲突

4. **SummaryGenerator 摘要生成**: 将相似信息聚合，生成带共识与分歧标注的社区摘要

5. **SearchOrchestrator 编排器**: 整合各组件，提供端到端可信搜索能力

该架构遵循**可扩展、可配置、可测试**的设计原则，便于根据实际需求进行调整和优化。
